 25         /**
 26          * Creates a new instance of class WebStore.
 27          * @access public
 28          * @constructs WebStore
 29          * @param {Storage} storageObject - Either sessionStorage or localStorage objects.
 30          * @param {string} storageId - The unique ID used as the key in which all record data are stored in the storageObject.
 31          * @param {WebStore~dataChangedCallback=} dataChanged - The callback that handles the data in storage changed.
 32          */
 33         function WebStore(storageObject, storageId, dataChanged) {
 34             // pre-condition
 35             console.assert(storageObject instanceof Storage, 'storageObject should be Storage');
 36             console.assert(typeof storageId === 'string', 'storageId should be string');
 37             if (dataChanged) {
 38                 console.assert(typeof dataChanged === 'function', 'dataChanged should be function');
 39             }
 40 
 41             /** 
 42              * Provides access to a list of key/value pairs, which are sometimes called items. 
 43              * Keys are strings. Any string (including the empty string) is a valid key. 
 44              * Values are similarly strings.
 45              * @access private
 46              * @member {Storage} 
 47              */
 48             this.storage = storageObject;
 49 
 50             /** 
 51              * The unique ID used as the key in which all record data are stored in the [storage]{@link WebStore#storage}.
 52              * @access private
 53              * @member {string} 
 54              */
 55             this.id = storageId;
 56 
 57             /** 
 58              * The callback that handles the data in [storage]{@link WebStore#storage} changed.
 59              * @access public
 60              * @member {WebStore~dataChangedCallback} 
 61              */
 62             this.dataChanged = dataChanged;
 63         }

 65         /**
 66          * When passed a key name and value, will add that key to the [storage]{@link WebStore#storage}, or update that key's value if it already exists.
 67          * @access protected
 68          * @memberof WebStore
 69          * @method
 70          * @param {string} key - A string containing the name of the key you want to create/update.
 71          * @param {*} value - The value you want to give the key you are creating/updating.
 72          * @returns No return value.
 73          */
 74         WebStore.prototype._setItem = function (key, value) {
 75             // pre-condition
 76             console.assert(typeof key === 'string', 'key should be string');
 77 
 78             this.storage.setItem(key, JSON.stringify(value));
 79         };

 81         /**
 82          * When passed a key name, will return that key's value.
 83          * @access protected
 84          * @memberof WebStore
 85          * @method
 86          * @param {string} key - A string containing the name of the key you want to retrieve the value of.
 87          * @returns {*} The value of the key. If the key does not exist, null is returned.
 88          */
 89         WebStore.prototype._getItem = function (key) {
 90             // pre-condition
 91             console.assert(typeof key === 'string', 'key should be string');
 92 
 93             var value = this.storage.getItem(key);
 94             return value ? JSON.parse(value) : null;
 95         };

105         WebStore.prototype._removeItem = function (key) {
106             // pre-condition
107             console.assert(typeof key === 'string', 'key should be string');

110         };

216          * Add a record id to the array of record IDs managed by this instance of WebStore.
217          * @access protected
218          * @memberof WebStore

220          * @returns No return value.

222         WebStore.prototype._addRecordId = function (recordId) {
223             // pre-condition
224             console.assert(typeof recordId === 'string', 'recordId should be string');
225

227             allRecordIds.push(recordId);

232          * Remove a record id from the array of record IDs managed by this instance of WebStore.
233          * @access protected
234          * @memberof WebStore
235          * @method
236          * @param {string} recordId - The ID of the record to remove.
237          * @returns No return value.
238          */
239         WebStore.prototype._removeRecordId = function (recordId) {
240             // pre-condition
241             console.assert(typeof recordId === 'string', 'recordId should be string');
242 
243             var allRecordIds = this.getAllRecordIds(),
244                 recordIndex = allRecordIds.indexOf(recordId);
245             if (recordIndex > -1) {
246                 allRecordIds.splice(recordIndex, 1);
247                 this._setItem(this.id, allRecordIds);
248             }

265         /**
266          * Create a record in the [storage]{@link WebStore#storage}.
267          * @access public
268          * @memberof WebStore
269          * @method
270          * @param {Object} record - The content of the created record.
271          * @returns {Object} The created record with "id" field initialized.
272          */
273         WebStore.prototype.createRecord = function (record) {
274             // pre-condition
275             console.assert(typeof record === 'object', 'record should be object');
276 
277             var newRecord = JSON.parse(JSON.stringify(record));
278             newRecord.id = this._incRecordCounterValue().toString();
279             this._setItem(this._getRecordKeyById(newRecord.id), newRecord);
280             this._addRecordId(newRecord.id);
281             this._incSize();
282             if (this.dataChanged) { this.dataChanged(); }
283             return newRecord;
284         };

343         /**
344          * The forEachRecord() method calls a provided function once for each record, in order.
345          * @access public
346          * @memberof WebStore
347          * @method     
348          * @param {WebStore~forEachRecordCallback} callback - A function to be run for each record.
349          * @param {number} [startIndex=0] - The index to start iterating at.
350          * @returns No return value.
351          */
352         WebStore.prototype.forEachRecord = function (callback, startIndex) {
353             console.assert(typeof callback === 'function', 'callback should be function');
354 
355             var allRecordIds = this.getAllRecordIds(),
356                 recordIndex,
357                 record;
358 
359             startIndex = typeof startIndex !== 'undefined' ? startIndex : 0;
360 
361             for (recordIndex = startIndex; recordIndex < allRecordIds.length; recordIndex += 1) {
362                 record = this.getRecordById(allRecordIds[recordIndex]);
363                 if (callback.bind(this)(record, recordIndex)) {
364                     break;
365                 }
366             }
367         };

384         /**
385          * Finds the index of the first matching record in this store by a specific field value.
386          * @access public
387          * @memberof WebStore
388          * @method
389          * @param {string} property - The name of the record field to test.
390          * @param {string|RegExp} value - Either a string that the field value should begin with, or a RegExp to test against the field.
391          * @param {number} [startIndex=0] - The index to start searching at.
392          * @returns {number} The matched index or -1.
393          */
394         WebStore.prototype.find = function (property, value, startIndex) {
395             var foundRecordIndex = -1,
396                 propertyValue;
397             this.forEachRecord(function (record, recordIndex) {
398                 if (record.hasOwnProperty(property)) {
399                     propertyValue = record[property];
400                     if (typeof propertyValue === 'string') {
401                         if (typeof value === 'string') {
402                             if (propertyValue.match(value)) {
403                                 foundRecordIndex = recordIndex;
404                                 return true;
405                             }
406                         }
407                     } else if (typeof propertyValue === 'number') {
408                         if (typeof value === 'number') {
409                             if (propertyValue === value) {
410                                 foundRecordIndex = recordIndex;
411                                 return true;
412                             }
413                         }
414                     }
415                 }
416             }, startIndex);
417             return foundRecordIndex;
418         };

432          * @access public

435          * @param {WebStore~findByCallback} fn - The function to be called.
436          * @param {number} [startIndex=0] - The index to start searching at.

444                 if (fn(record, recordIndex)) {
445                     foundRecordIndex = recordIndex;
446                     return true;
447                 }
448             }, startIndex);

495          * The getDhtmlxGrid method will return the formatted object to be loaded into the DHTMLX grid.
496          * @access public
497          * @memberof WebStore
498          * @method
499          * @returns {DhtmlxGridJsonDataStructure} An object containing the values for the dhtmlxGrid.

520         /**
521          * The getDhtmlxGridRowByRecordId method is exactly that. 
522          * This will structure a record object into a formatted row item for use in the grid data.
523          * @access public
524          * @memberof WebStore
525          * @method
526          * @returns {DhtmlxGridRowJsonDataStructure} An object for record row.
527          */
528         WebStore.prototype.getDhtmlxGridRowByRecordId = function (recordId, columnKeys) {
529             // pre-condition
530             console.assert(typeof recordId === 'string', 'recordId should be string');
531 
532             var record = this.getRecordById(recordId), data = [];
533             if (typeof columnKeys === "undefined") {
534                 columnKeys = Object.keys(record);
535             }
536             columnKeys.forEach(function (key) {
537                 if (record.hasOwnProperty(key)) {
538                     data.push(record[key]);
539                 }
540             });
541             return {
542                 "id" : record.id,
543                 "data" : data
544             };
545         };
